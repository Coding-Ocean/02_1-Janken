#include"libOne.h"
void gmain(){
    window(800, 450);
    //画像読み込み
    int playerGuImg = loadImage("assets\\playerGu.png");
    int playerChokiImg = loadImage("assets\\playerChoki.png");
    int playerPaImg = loadImage("assets\\playerPa.png");
    int pcGuImg = loadImage("assets\\pcGu.png");
    int pcChokiImg = loadImage("assets\\pcChoki.png");
    int pcPaImg = loadImage("assets\\pcPa.png");
    //手の番号
    int GU = 0;
    int CHOKI = 1;
    int PA = 2;
    //プレイヤーデータ
    int playerImg = playerGuImg;
    float playerPx = 225;
    float playerPy = 225;
    //ＰＣデータ
    int pcImg = pcGuImg;
    int pcHand = GU;
    float pcPx = 575;
    float pcPy = 225;
    //メインループ
    while (notQuit){
        //A,S,Dいずれかのキーが押されたら
        if (isTrigger(KEY_A) || isTrigger(KEY_S) || isTrigger(KEY_D)) {
            //プレイヤーの手を決める
            if (isTrigger(KEY_A)) {
                playerImg = playerGuImg;
            }
            else if (isTrigger(KEY_S)) {
                playerImg = playerChokiImg;
            }
            else if (isTrigger(KEY_D)) {
                playerImg = playerPaImg;
            }
            //ＰＣの手を決める
            pcHand = random() % 3;
            if (pcHand == GU) {
                pcImg = pcGuImg;
            }
            else if (pcHand == CHOKI) {
                pcImg = pcChokiImg;
            }
            else if (pcHand == PA) {
                pcImg = pcPaImg;
            }
        }
        clear(200);
        rectMode(CENTER);
        image(playerImg, playerPx, playerPy);
        image(pcImg, pcPx, pcPy);
    }
}
